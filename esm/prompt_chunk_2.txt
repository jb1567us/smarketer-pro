async function processMassBatch() {
        if (typeof wp === 'undefined' || !wp.data) {
            console.error("wp.data not available");
            return "Error: wp.data missing";
        }

        const stats = { created: 0, updated: 0, skipped: 0, errors: 0 };
        const log = [];

        console.log(`Starting mass creation for ${artworkData.length} items...`);

        // Chunking to avoid rate limits
        const chunkSize = 1; // 1 item per batch to be safe and observe logs
        // Actually, let's do 1 but run fast.

        for (const item of window.MASS_DATA) {
            try {
                // Enrich data
                const title = item.cleanTitle || item.title.replace(/ Painting$/i, '');
                // Handle "cleanTitle" if missing in JSON (some failed items might have it or not)
                const finalTitle = title;

                const slug = item.slug ? item.slug.replace(/Painting$/i, '') : finalTitle.toLowerCase().replace(/[^a-z0-9]+/g, '-');
                const price = item.price || DEFAULT_PRICE;
                // Construct dims string
                let dims = DEFAULT_DIMS;
                if (item.width && item.height) {
                    dims = `${item.height}" × ${item.width}"`; // H x W
                } else if (item.dimensions) {
                    dims = item.dimensions;
                }

                const medium = item.medium || DEFAULT_MEDIUM;
                const year = item.year || DEFAULT_YEAR;
                const framing = DEFAULT_FRAMING;
                const colorDesc = DEFAULT_COLOR_DESC;

                // Tags
                const tagsDisplay = `${medium}, Abstract, Contemporary`;
                const tagsData = `data-style="abstract" data-medium="${medium.toLowerCase()}"`;

                // Prepare Content
                let content = window.layoutTemplate
                    .replace(/{{TITLE}}/g, finalTitle)
                    .replace(/{{IMAGE_URL}}/g, item.image_url) // We have image_url from the initial scan!
                    .replace(/{{PRICE}}/g, price)
                    .replace(/{{MEDIUM}}/g, medium)
                    .replace(/{{DIMENSIONS}}/g, dims)
                    .replace(/{{YEAR}}/g, year)
                    .replace(/{{COLOR_DESC}}/g, colorDesc)
                    .replace(/{{FRAMING}}/g, framing)
                    .replace(/{{TAGS_DISPLAY}}/g, tagsDisplay)
                    .replace(/{{TAGS_DATA}}/g, tagsData)
                    .replace(/{{SAATCHI_URL}}/g, item.saatchi_url);

                // Check existing page
                console.log(`Processing: ${finalTitle}`);

                const existingPages = await wp.apiFetch({ path: `/wp/v2/pages?search=${encodeURIComponent(finalTitle)}` });
                let pageIdToUpdate = null;
                if (existingPages && existingPages.length > 0) {
                    const match = existingPages.find(p => p.title.rendered === finalTitle);
                    if (match) pageIdToUpdate = match.id;
                }

                if (pageIdToUpdate) {
                    console.log(`Updating existing page ${pageIdToUpdate}...`);
                    await wp.apiFetch({
                        path: `/wp/v2/pages/${pageIdToUpdate}`,
                        method: 'PUT',
                        data: {
                            content: content
                            // Don't change slug for existing pages to keep links working? 
                            // Actually, user wants "clean URL structure". 
                            // But usually best to preserve if it exists.
                        }
                    });
                    stats.updated++;
                } else {
                    console.log(`Creating new page...`);
                    const newPage = await wp.apiFetch({
                        path: '/wp/v2/pages',
                        method: 'POST',
                        data: {
                            title: finalTitle,
                            content: content,
                            status: 'publish',
                            slug: slug
                        }
                    });
                    console.log(`Created: ${newPage.link}`);
                    stats.created++;
                }

                // Wait 1s between items
                await new Promise(r => setTimeout(r, 1000));

            } catch (err) {
                console.error(`Error processing ${item.title}:`, err);
                stats.errors++;
            }
        }

        console.log("Mass Creation Complete:", stats);
        console.log("DATA_START");
        console.log(JSON.stringify(stats));
        console.log("DATA_END");
    }
